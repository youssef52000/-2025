  function parseQuestions(compressedList) {
    return compressedList.map(line => {
      const parts = line.split('|');
      return {
        category: parts[0],
        q: parts[1],
        correct: parts[2],
        options: parts.slice(3)
      };
    });
  }


  const allQuestions = parseQuestions(COMPRESSED_QUESTIONS);
  const categories = [...new Set(allQuestions.map(q => q.category))].sort();
  let score = 0;
  let correctCount = 0;
  let wrongCount = 0;
  let currentCategory = null;
  let answeredCount = 0;
  let totalInCategory = 0;
  let consecutiveCorrect = 0;
  let totalCorrect = 0;
  let totalQuestions = allQuestions.length;


  const quizContainer = document.getElementById('quiz');
  const catButtonsContainer = document.getElementById('categoryButtons');
  const statsEl = document.getElementById('stats');
  const progressFill = document.getElementById('progressFill');
  const congratsModal = document.getElementById('congratsModal');
  const congratsText = document.getElementById('congratsText');
  const scoreSpan = document.getElementById('scoreSpan');


  function updateStats() {
    statsEl.innerHTML = `النقاط: ${score} | الإجابات الصحيحة: ${correctCount} | الخاطئة: ${wrongCount}`;
  }


  function updateProgress() {
    if (totalInCategory > 0) {
      const percent = (answeredCount / totalInCategory) * 100;
      progressFill.style.width = percent + '%';
    }
  }


  function showCongrats() {
    congratsModal.classList.remove('hidden');
    const percentage = (correctCount / totalInCategory) * 100;
    let msg = "أحسنت! أداء ممتاز!";
    if (percentage >= 80) msg = "ممتاز! أداء مذهل!";
    else if (percentage >= 60) msg = "جيد جدًا! أداء جيد!";
    else if (percentage >= 40) msg = "مقبول! حاول أكثر!";
    else msg = "تحتاج للمزيد من التدريب!";
    congratsText.textContent = msg;
    scoreSpan.textContent = score;
  }


  function closeCongrats() {
    congratsModal.classList.add('hidden');
  }


  function showMoreCategories() {
    closeCongrats();
    openMore();
  }


  function shuffle(arr) {
    for (let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
  }


  function buildCategoryButtons() {
    catButtonsContainer.classList.add('controls-grid');
    catButtonsContainer.innerHTML = '';
    const visibleCategories = categories.slice(0, 6);
    const moreBtn = document.createElement('button');
    moreBtn.className = 'more-btn';
    moreBtn.innerHTML = '<i class="fas fa-ellipsis-h"></i> المزيد';
    moreBtn.onclick = () => { openMore(); };
    visibleCategories.forEach(cat => {
      const b = document.createElement('button');
      b.innerHTML = `<i class="fas fa-star"></i><div class="btn-text">${cat}</div>`;
      b.onclick = () => { renderQuestions(cat); setActiveButton(b); };
      catButtonsContainer.appendChild(b);
    });
    catButtonsContainer.appendChild(moreBtn);
  }


  function openMore() {
    const panel = document.getElementById('morePanel');
    const grid = document.getElementById('moreCategories');
    grid.innerHTML = '';
    const moreCats = categories.slice(6);
    moreCats.forEach(cat => {
      const b = document.createElement('button');
      b.textContent = cat;
      b.onclick = () => {
        renderQuestions(cat);
        setActiveButton(b);
        closeMore();
      };
      grid.appendChild(b);
    });
    panel.style.display = 'flex';
  }


  function closeMore() {
    document.getElementById('morePanel').style.display = 'none';
  }


  function setActiveButton(btn) {
    Array.from(catButtonsContainer.children).forEach(c => c.classList.remove('active'));
    btn.classList.add('active');
  }


  function renderQuestions(category = null) {
    currentCategory = category;
    quizContainer.innerHTML = '';
    let items = allQuestions.slice();
    if (category) items = items.filter(q => q.category === category);
    totalInCategory = items.length;
    answeredCount = 0;
    updateProgress();
    statsEl.textContent = `عرض ${items.length} سؤال${items.length !== 1 ? 's' : ''} ${category ? 'في ' + category : '(كل الفئات)'}`;
    
    const grouped = {};
    items.forEach(q => {
      if (!grouped[q.category]) grouped[q.category] = [];
      grouped[q.category].push(q);
    });


    Object.keys(grouped).sort().forEach(cat => {
      const catDiv = document.createElement('div');
      catDiv.className = 'category';
      catDiv.innerHTML = `<div class="cat-title">${cat} <span class="small">(${grouped[cat].length} سؤال)</span></div>`;
      quizContainer.appendChild(catDiv);


      grouped[cat].forEach(qObj => {
        const div = document.createElement('div');
        div.className = 'question';
        const opts = qObj.options.slice();
        shuffle(opts);
        const correctIndex = opts.findIndex(o => o === qObj.correct);
        let finalCorrectIndex = correctIndex;
        if (finalCorrectIndex === -1) {
          finalCorrectIndex = Math.floor(Math.random() * opts.length);
          opts[finalCorrectIndex] = qObj.correct;
        }
        const optButtonsHtml = opts.map((opt, i) => {
          const label = ['أ', 'ب', 'ج', 'د'][i];
          return `<button data-ans="${i}" onclick="handleAnswer(this, ${finalCorrectIndex})">${label}) ${opt}</button>`;
        }).join('');
        div.innerHTML = `<div class="q-text"><strong>${qObj.category} —</strong> ${qObj.q}</div><div class="options">${optButtonsHtml}</div>`;
        quizContainer.appendChild(div);
      });
    });
  }


  window.handleAnswer = function(btn, correctIndex) {
    if (btn.disabled) return;
    const parent = btn.parentElement;
    const selected = Number(btn.getAttribute('data-ans'));
    const children = Array.from(parent.children);


    if (selected === correctIndex) {
      btn.classList.add('correct');
      score += 10;
      correctCount++;
      consecutiveCorrect++;
      totalCorrect++;
      // تحقق من التهاني
      if ([10, 20, 30, 40, 100, 500].includes(consecutiveCorrect)) {
        alert(`ممتاز! ${consecutiveCorrect} إجابة صحيحة متتالية!`);
      }
    } else {
      btn.classList.add('wrong');
      const corrBtn = children[correctIndex];
      if (corrBtn) corrBtn.classList.add('correct');
      wrongCount++;
      consecutiveCorrect = 0;
    }


    children.forEach(b => b.disabled = true);
    answeredCount++;
    updateStats();
    updateProgress();


    if (currentCategory && answeredCount === totalInCategory) {
      showCongrats();
    }
    if (totalCorrect === totalQuestions) {
      congratsText.textContent = "أحسنت! لقد أجبت على كل الأسئلة بشكل صحيح!";
      scoreSpan.textContent = score;
      congratsModal.classList.remove('hidden');
    }
  };


  // ✅ إعدادات الموقع
  function toggleSettings() {
    const panel = document.getElementById('settingsPanel');
    panel.style.display = panel.style.display === 'flex' ? 'none' : 'flex';
  }


  document.getElementById('darkModeToggle').addEventListener('change', function() {
    document.body.classList.toggle('dark-mode', this.checked);
  });


  function resetSettings() {
    document.documentElement.style.setProperty('--bg-color', '#f5f7fa');
    document.documentElement.style.setProperty('--accent-color', '#3b82f6');
    document.getElementById('darkModeToggle').checked = false;
    document.body.classList.remove('dark-mode');
  }


  buildCategoryButtons();
  renderQuestions();
